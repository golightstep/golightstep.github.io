SDK Reference

Quickstart

from superbrain import DistributedContextFabric fabric = DistributedContextFabric() # Connect to fabric ctx = fabric.attach_context("my-context") # Attach/create a context ctx.write("key1", "Hello from Python!") # Write data value = ctx.read("key1") # Read data keys = ctx.list_keys() # List all keys print(value)

API Reference

Method Description
DistributedContextFabric(coordinator="localhost:60050") Connect to the SuperBrain cluster
fabric.attach_context(context_id: str) Attach to (or create) a named shared context
ctx.write(key: str, value: str) Write a value to the distributed fabric
ctx.read(key: str) → str Read a value by key
ctx.list_keys() → list[str] List all keys in this context

Code Example — CrewAI Shared Memory

from superbrain import DistributedContextFabric import json, time class SuperBrainMemory: def __init__(self, context_id="collective_intelligence"): self.fabric = DistributedContextFabric() self.ctx = self.fabric.attach_context(context_id) def save(self, content, metadata): key = f"mem_{time.time_ns()}" payload = json.dumps({"body": content, "meta": metadata, "ts": time.time()}) self.ctx.write(key, payload) def search(self): keys = self.ctx.list_keys() return [json.loads(self.ctx.read(k)) for k in keys if self.ctx.read(k)] # Agent A writes, Agent B reads — even across different processes mem = SuperBrainMemory() mem.save("SuperBrain latency is 13.5μs", {"source": "agent_a"}) history = mem.search() for item in history: print(item['body'])

const { Client } = require('superbrain-distributed-sdk'); const fabric = new Client('localhost:60050'); // Connect to fabric // Write data const data = Buffer.from(JSON.stringify({ msg: "Hello from Node!" })); await fabric.write("key1", 0, data); // Read data const raw = await fabric.read("key1", 0, 1024); const value = JSON.parse(raw.toString()); console.log(value.msg);

API Reference

Method Description
new Client(address: string) Connect to the SuperBrain coordinator
fabric.write(ptrId, offset, data: Buffer) Write a Buffer to the fabric at a pointer
fabric.read(ptrId, offset, size) → Buffer Read bytes from a pointer

Code Example — Multi-Agent Collective Memory

const { Client } = require('superbrain-distributed-sdk'); async function main() { const fabric = new Client('localhost:60050'); // Agent Alpha: writes a discovery const payload = Buffer.from(JSON.stringify({ body: "Fan-out latency verified at 13.5μs", source: "Agent Alpha", timestamp: new Date().toISOString() })); await fabric.write("latest_discovery", 0, payload); console.log("✅ Agent Alpha: Persisted to fabric."); // Agent Beta: reads it independently const raw = await fabric.read("latest_discovery", 0, 1024); const memory = JSON.parse(raw.toString()); console.log(`📥 Agent Beta Recalled: "${memory.body}"`); } main().catch(console.error);

API Reference

Method Description
sdk.NewClient(coordinator string) Connect to the SuperBrain cluster
client.Register(nodeID string) Enroll node in the fabric (sets up mTLS)
client.Allocate(size int64) *Pointer Allocate shared memory, returns a Pointer
client.Write(ptr, offset, data []byte) Write bytes to the fabric
client.Read(ptr, offset, size) []byte Read bytes from the fabric

Code Example — High-Frequency Market Feed

package main import ( "fmt" "time" "github.com/golightstep/superbrainSdk/pkg/sdk" ) func main() { client, _ := sdk.NewClient("localhost:60050") client.Register("go-feed-node") // Allocate 4MB shared pointer for market data ptr, _ := client.Allocate(4 * 1024 * 1024) fmt.Printf("🚀 Feed initialized: %s\n", ptr.ID) // Writer goroutine (simulates market ticks) go func() { for i := 0; ; i++ { tick := []byte(fmt.Sprintf(`{"symbol":"SBUSD","price":13.5,"ts":%d}`, time.Now().UnixNano())) client.Write(ptr, 0, tick) if i%100 == 0 { fmt.Printf("📈 Published %d ticks\n", i) } time.Sleep(100 * time.Millisecond) } }() // Reader: a different strategy consuming the same shared memory for { data, err := client.Read(ptr, 0, 1024) if err == nil && len(data) > 0 { fmt.Printf("📥 Latest tick: %s\n", string(data)) } time.Sleep(500 * time.Millisecond) } }
